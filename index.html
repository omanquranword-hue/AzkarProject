<script>

    // وظيفة عرض الأذكار في الصفحة الرئيسية

    function loadMainAzkar() {

        const container = document.getElementById('main-azkar-container');

        const azkar = JSON.parse(localStorage.getItem('azkarData')) || [];

        

        if (azkar.length === 0) {

            container.innerHTML = '<p>لا توجد أذكار مضافة، انتقل للإدارة لإضافتها.</p>';

            return;

        }


        container.innerHTML = '';

        azkar.forEach((item, index) => {

            container.innerHTML += `

                <div class="zekr-card" onclick="decrementCount(${index})">

                    <span class="zekr-text">${item.text}</span>

                    <div class="counter" id="count-${index}">${item.count}</div>

                </div>

            `;

        });

    }


    // وظيفة تقليل العداد عند النقر (للاستخدام على الهاتف)

    function decrementCount(index) {

        let azkar = JSON.parse(localStorage.getItem('azkarData'));

        if (azkar[index].count > 0) {

            azkar[index].count--;

            document.getElementById(`count-${index}`).innerText = azkar[index].count;

            

            // إذا وصل للصفر يمكن تمييزه بلون مختلف

            if (azkar[index].count === 0) {

                document.getElementById(`count-${index}`).parentElement.style.opacity = "0.5";

            }

        }

    }


    window.onload = loadMainAzkar;

</script>
